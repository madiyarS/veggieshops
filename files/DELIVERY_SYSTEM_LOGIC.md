## 2. АЛГОРИТМ ПРОВЕРКИ ДОСТАВКИ

### 2.1 Проверка доступности доставки (на frontend & backend)

```
ВХОД: customer_address, selected_store_id

1. Получить координаты магазина (store.latitude, store.longitude)

2. Получить координаты адреса клиента
   a) Если в списке улиц магазина → получить усредненные координаты района
   b) Если текстовое поле → примерные координаты (или skip)

3. Рассчитать расстояние между магазином и адресом:
   distance = haversine(store.lat, store.lon, customer.lat, customer.lon)
   
   Формула Хаверсина на Go:
   ────────────────────────
   func haversine(lat1, lon1, lat2, lon2 float64) float64 {
       const R = 6371 // Радиус Земли в км
       dlat := (lat2 - lat1) * math.Pi / 180
       dlon := (lon2 - lon1) * math.Pi / 180
       a := math.Sin(dlat/2)*math.Sin(dlat/2) +
           math.Cos(lat1*math.Pi/180)*math.Cos(lat2*math.Pi/180)*
           math.Sin(dlon/2)*math.Sin(dlon/2)
       c := 2 * math.Atan2(math.Sqrt(a), math.Sqrt(1-a))
       return R * c
   }

4. Проверить: distance <= store.delivery_radius_km?
   ├─ ДА: Доставка доступна ✓
   └─ НЕТ: Показать ошибку "В вашем районе нет доставки" ✗

5. Если доставка доступна → найти соответствующий район
   selected_district = find_district(customer_address, store_id)
   
6. Если район найден → получить стоимость доставки
   delivery_fee = selected_district.delivery_fee_regular (или express)

ВЫХОД: {is_available: bool, district_id: UUID, delivery_fee: int}
```

### 2.2 Визуальная схема проверки

```
┌─ Клиент вводит адрес доставки
│
├─ Система находит координаты адреса
│
├─ Рассчитывает расстояние до магазина
│
└─→ Расстояние <= 3км? ─┬─→ ДА → Показать типы доставки
                        │      (Regular & Express)
                        │
                        └─→ НЕТ → Ошибка: нет доставки
```

## 5. ИНТЕГРАЦИЯ С КАРТАМИ (ОПЦИОНАЛЬНО)

### 5.1 Текущий вариант (БЕЗ карт)

```
✓ Решение 1: Выбор из списка районов
   └─ Просто, быстро, работает офлайн
   
✓ Решение 2: Выбор улицы из выпадающего списка
   └─ Администратор добавляет улицы в админ-панели
   
✗ Минус: Клиент не видит реальное местоположение магазина/района
```

### 5.2 Будущий вариант (С картами)

**Интеграция Google Maps / 2GIS / Yandex.Maps:**

```javascript
// Frontend: React компонент с картой

import { GoogleMap, LoadScript, Marker } from '@react-google-maps/api';

function DeliveryMap() {
  const [storeLocation] = useState({
    lat: 51.1694,  // Астана
    lng: 71.4491
  });
  
  const [customerLocation, setCustomerLocation] = useState(null);
  
  const onMapClick = (e) => {
    const lat = e.latLng.lat();
    const lng = e.latLng.lng();
    
    // Проверяем расстояние
    const distance = calculateDistance(storeLocation, { lat, lng });
    
    if (distance <= DELIVERY_RADIUS) {
      setCustomerLocation({ lat, lng });
      // Запрашиваем стоимость доставки
      getDeliveryFee(lat, lng);
    } else {
      showError('Доставка недоступна в этом районе');
    }
  };
  
  return (
    <LoadScript googleMapsApiKey={process.env.REACT_APP_GOOGLE_MAPS_KEY}>
      <GoogleMap
        mapContainerStyle={{ height: '400px', width: '100%' }}
        center={storeLocation}
        zoom={13}
        onClick={onMapClick}
      >
        {/* Маркер магазина */}
        <Marker
          position={storeLocation}
          title="Наш магазин"
          icon={{
            path: google.maps.SymbolPath.CIRCLE,
            scale: 15,
            fillColor: '#4CAF50',
            fillOpacity: 1,
            strokeColor: '#2E7D32',
            strokeWeight: 2,
          }}
        />
        
        {/* Маркер адреса клиента */}
        {customerLocation && (
          <Marker position={customerLocation} title="Ваш адрес" />
        )}
        
        {/* Круг радиуса доставки */}
        <Circle
          center={storeLocation}
          radius={DELIVERY_RADIUS * 1000} // в метрах
          options={{
            fillColor: 'rgba(76, 175, 80, 0.1)',
            strokeColor: '#4CAF50',
            strokeWeight: 2,
          }}
        />
      </GoogleMap>
    </LoadScript>
  );
}
```

## 8. DATABASE QUERIES

### 8.1 Проверка доступности доставки (SQL)

```sql
-- Получить магазины в пределах доставки для адреса клиента
SELECT 
    s.id,
    s.name,
    s.latitude,
    s.longitude,
    s.delivery_radius_km,
    s.min_order_amount,
    -- Формула для расстояния (PostGIS или базовая математика)
    (6371 * acos(cos(radians(51.1700)) * cos(radians(s.latitude)) * 
     cos(radians(s.longitude) - radians(71.4500)) + 
     sin(radians(51.1700)) * sin(radians(s.latitude)))) AS distance_km
FROM stores s
WHERE (6371 * acos(cos(radians(51.1700)) * cos(radians(s.latitude)) * 
       cos(radians(s.longitude) - radians(71.4500)) + 
       sin(radians(51.1700)) * sin(radians(s.latitude)))) <= s.delivery_radius_km
AND s.is_active = true
ORDER BY distance_km ASC;

-- Получить диагнозы для магазина
SELECT 
    id,
    name,
    distance_km,
    delivery_fee_regular,
    delivery_fee_express
FROM districts
WHERE store_id = $1
AND is_active = true
ORDER BY name;

-- Получить улицы в районе
SELECT street_name
FROM district_streets
WHERE district_id = $1
ORDER BY street_name;
```

### 8.2 Создание заказа с проверкой доставки

```sql
-- Транзакция для создания заказа
BEGIN;

-- 1. Проверяем наличие товара на складе
SELECT stock_quantity FROM products WHERE id = $1 FOR UPDATE;

-- 2. Проверяем доступность района
SELECT * FROM districts WHERE id = $1 AND is_active = true;

-- 3. Проверяем доступность временного окна
SELECT max_orders - COUNT(*) as available_slots
FROM orders
WHERE delivery_time_slot_id = $1 AND status != 'cancelled'
```
